conda环境的命令不会重复执行,这时我们最后手动再复制一次即可:
```bash
# >>> conda initialize >>>
# !! Contents within this block are managed by 'conda init' !!
__conda_setup="$('/mnt/project_modelware_roce/zhaojian/cyun/app/miniconda3/bin/conda' 'shell.bash' 'hook' 2> /dev/null)"
if [ $? -eq 0 ]; then
    eval "$__conda_setup"
else
    if [ -f "/mnt/project_modelware_roce/zhaojian/cyun/app/miniconda3/etc/profile.d/conda.sh" ]; then
        . "/mnt/project_modelware_roce/zhaojian/cyun/app/miniconda3/etc/profile.d/conda.sh"
    else
        export PATH="/mnt/project_modelware_roce/zhaojian/cyun/app/miniconda3/bin:$PATH"
    fi
fi
unset __conda_setup
# <<< conda initialize <<<
```
